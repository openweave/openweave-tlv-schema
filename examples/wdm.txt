wdm-resource-id => CHOICE OF
(
  general                              : BYTE STRING [len 10],
  device-id                            : UNSIGNED INTEGER [range 64bits]
)

wdm-locator => STRUCTURE [any-order]
{
  resource-id [1,optional]             : wdm-resource-id,
  profile-id [2]                       : CHOICE OF 
    (
      id-only                          : UNSIGNED INTEGER [range 32bits],
      id-and-versions                  : ARRAY
        (
          id                           : UNSIGNED INTEGER [range 32bits],
          max-version                  : UNSIGNED INTEGER [range 16bits],
          min-version                  : UNSIGNED INTEGER [range 16bits]
        )
    ),
  instance-id [3,optional]             : UNSIGNED INTEGER [range 64bits]
}

wdm-path => LIST
(
  locator [1]                          : wdm-locator,
                                         ANY *
)
